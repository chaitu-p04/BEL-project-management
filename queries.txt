CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  username VARCHAR(50) NOT NULL,
  staff_id VARCHAR(20) UNIQUE NOT NULL,
  password VARCHAR(200) NOT NULL
);

CREATE TABLE projects (
  id SERIAL PRIMARY KEY,
  project_name VARCHAR(255),
  start_date DATE,
  end_date DATE,
  customer_name VARCHAR(255),
  customer_address TEXT,
  customer_contact TEXT,
  order_value NUMERIC,
  duration VARCHAR(100),
  kickoff_date DATE,
  kickoff_status TEXT
);
CREATE TABLE project_sales (
  id SERIAL PRIMARY KEY,
  project_id INTEGER REFERENCES projects(id),
  order_acceptance TEXT,
  customer_indemnity TEXT,
  customer_pbg TEXT,
  contract_creation TEXT,
  contract_approval TEXT,
  sale_order_creation TEXT,
  sale_order_release TEXT,
  contract_number_date TEXT,
  sale_order_number_date TEXT,
  spare1 TEXT,
  spare2 TEXT
);
CREATE TABLE project_stakeholders (
  id SERIAL PRIMARY KEY,
  project_id INTEGER REFERENCES projects(id),
  project_manager TEXT,
  team_members TEXT,
  architect_team TEXT,
  sales_team TEXT,
  qa_team TEXT,
  purchase_team TEXT,
  finance_team TEXT
);
